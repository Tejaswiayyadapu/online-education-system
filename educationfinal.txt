<script>
var messages = new Array
(
   'EDUCATION TYPES',
   'EDUCATION TYPES',
   'EDUCATION TYPES'
);
var currentMsg = 0;
var fontSize = 0;
var fontSizeMax = 72;
var delay = 2;
var speed = 12;

function scaleText() 
{
   if (fontSize < fontSizeMax) 
   {
      var text_zoomer = document.getElementById("text_zoomer");
      text_zoomer.style.fontSize = fontSize + "px";
      fontSize++;
      setTimeout(function(){scaleText();}, speed);
    }
}
function zoomText() 
{
   if (currentMsg >= messages.length) 
   {
      currentMsg = 0;
   }
   var text_zoomer = document.getElementById("text_zoomer");
   text_zoomer.innerHTML = messages[currentMsg];
   fontSize = 0;
   scaleText();
   currentMsg++;
   setTimeout(function(){zoomText();}, delay*1000);
}
zoomText();
